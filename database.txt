##customer table
CREATE TABLE Customer (
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL UNIQUE,
    phone_num VARCHAR(10) NOT NULL,
    phone_num2 VARCHAR(10),
    address TEXT,
    price_range BIGINT,
    start_date DATE,
    place VARCHAR(200),
    service VARCHAR(255) NOT NULL,
    message TEXT,
    status_c VARCHAR(50) DEFAULT 'not selected',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY(email)
);

##completed table
CREATE TABLE Compro (
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) NOT NULL,
    phone_num VARCHAR(10) NOT NULL UNIQUE,
    phone_num2 VARCHAR(10),
    address TEXT NOT NULL,
    Profit_price DECIMAL(12,2),
    start_date DATE,
    ending_date DATE,
    place VARCHAR(200),
    service VARCHAR(255) NOT NULL,
    message TEXT,
    comp_pro_price DECIMAL(12,2),
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    PRIMARY KEY (email)
);
